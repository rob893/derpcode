pub struct Solution;

impl Solution {
    pub fn fib(n: i32) -> i32 {
        if n <= 1 {
            return n;
        }
        
        let mut prev2 = 0; // F(0)
        let mut prev1 = 1; // F(1)
        
        for _ in 2..=n {
            let current = prev1 + prev2;
            prev2 = prev1;
            prev1 = current;
        }
        
        prev1
    }
}
