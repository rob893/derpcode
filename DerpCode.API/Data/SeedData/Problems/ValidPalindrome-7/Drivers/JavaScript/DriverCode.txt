import { BaseDriver, IProblemDriver } from './base-driver.js';
import { isPalindrome } from './solution.js';

/**
 * Problem-specific driver for ValidPalindrome problem.
 */
class ValidPalindromeDriver extends IProblemDriver {
    /**
     * Parse input as array of strings and expected output as array of booleans.
     */
    parseTestCases(input, expectedOutput) {
        const testCases = [];
        
        for (let i = 0; i < input.length; i++) {
            testCases.push({
                input: input[i],
                expectedOutput: expectedOutput[i]
            });
        }
        
        return testCases;
    }

    /**
     * Execute the isPalindrome function with the test case input.
     */
    executeTestCase(testCase) {
        return isPalindrome(testCase.input);
    }

    /**
     * Compare results using boolean equality.
     */
    compareResults(actual, expected) {
        return actual === expected;
    }

    /**
     * Format error message for failed tests.
     */
    formatErrorMessage(actual, expected) {
        return `Expected ${expected} but got ${actual}`;
    }
}

// Create and run the driver
const driver = new BaseDriver(new ValidPalindromeDriver());
driver.run();
