import java.util.*;
import com.google.gson.JsonArray;
import com.google.gson.JsonElement;
import com.google.gson.JsonParser;

/**
 * Problem-specific driver for ValidPalindrome problem.
 */
class ValidPalindromeDriver extends ProblemDriverBase {
    @Override
    public List<TestCase> parseTestCases(JsonArray input, JsonArray expectedOutput) {
        List<TestCase> testCases = new ArrayList<>();
        
        for (int i = 0; i < input.size(); i++) {
            String s = input.get(i).getAsString();
            boolean expected = expectedOutput.get(i).getAsBoolean();
            
            testCases.add(new TestCase(s, expected));
        }
        
        return testCases;
    }

    @Override
    public Object executeTestCase(TestCase testCase, int index) throws Exception {
        String input = (String) testCase.getInput();
        Solution solution = new Solution();
        return solution.isPalindrome(input);
    }

    @Override
    public boolean compareResults(Object actual, Object expected) {
        return Objects.equals(actual, expected);
    }
}

class Program {
    public static void main(String[] args) {
        // Create and run the driver
        BaseDriver driver = new BaseDriver(new ValidPalindromeDriver());
        driver.run(args);
    }
}
