def is_valid(s: str) -> bool:
    stack = []
    pairs = {
        ')': '(',
        '}': '{',
        ']': '['
    }
    
    for char in s:
        if char in '({[':
            stack.append(char)
        else:
            if not stack:
                return False
            
            top = stack.pop()
            if top != pairs[char]:
                return False
    
    return len(stack) == 0
