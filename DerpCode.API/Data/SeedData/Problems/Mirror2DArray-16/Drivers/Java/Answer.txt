import java.util.Arrays;

public class Solution {
    public static int[][] mirror2DArray(int[][] matrix) {
        int[][] result = new int[matrix.length][];

        for (int i = 0; i < matrix.length; i++) {
            int[] rowCopy = Arrays.copyOf(matrix[i], matrix[i].length);
            for (int left = 0, right = rowCopy.length - 1; left < right; left++, right--) {
                int temp = rowCopy[left];
                rowCopy[left] = rowCopy[right];
                rowCopy[right] = temp;
            }
            result[i] = rowCopy;
        }

        return result;
    }
}
