import { ProblemDriverBase, BaseDriver, TestCase } from './base-driver';
import { maxSubArray } from './solution';

class MaximumSubarrayDriver extends ProblemDriverBase {
    parseTestCases(input: any, expectedOutput: any): TestCase[] {
        return input.map((arr: number[], index: number) => ({
            input: arr,
            expectedOutput: expectedOutput[index]
        }));
    }

    executeTestCase(testCase: TestCase, index: number): any {
        return maxSubArray(testCase.input as number[]);
    }

    compareResults(actual: any, expected: any): boolean {
        return actual === expected;
    }
}

const driver = new BaseDriver(new MaximumSubarrayDriver());
driver.run();
