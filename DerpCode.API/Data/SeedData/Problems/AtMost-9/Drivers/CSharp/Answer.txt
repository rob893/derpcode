using System;

public class Solution
{
    public static bool AtMost(int[] arr, int target, int atMostNTimes)
    {
        int count = 0;
        
        foreach (int num in arr)
        {
            if (num == target)
            {
                count++;
                if (count > atMostNTimes)
                {
                    return false;
                }
            }
        }
        
        return true;
    }
}
