/**
 * Determines if the target appears in the array at most atMostNTimes times.
 * @param arr - The array to search in
 * @param target - The number to count
 * @param atMostNTimes - The maximum allowed occurrences
 * @returns True if target appears <= atMostNTimes, false otherwise
 */
export function atMost(arr: number[], target: number, atMostNTimes: number): boolean {
    let count = 0;
    
    for (const num of arr) {
        if (num === target) {
            count++;
            if (count > atMostNTimes) {
                return false;
            }
        }
    }
    
    return true;
}
