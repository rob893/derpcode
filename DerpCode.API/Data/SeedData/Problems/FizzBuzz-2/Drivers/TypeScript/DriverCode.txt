import { ProblemDriverBase, BaseDriver, TestCase } from './base-driver';
import { fizzBuzz } from './solution';

/**
 * Problem-specific driver for FizzBuzz problem.
 */
class FizzBuzzDriver extends ProblemDriverBase {
    /**
     * Parse input as array of integers and expected output as array of strings.
     */
    parseTestCases(input: any, expectedOutput: any): TestCase[] {
        const testCases: TestCase[] = [];
        
        for (let i = 0; i < input.length; i++) {
            testCases.push({
                input: input[i],
                expectedOutput: expectedOutput[i]
            });
        }
        
        return testCases;
    }

    /**
     * Execute the fizzBuzz function with the test case input.
     */
    executeTestCase(testCase: TestCase, index: number): any {
        return fizzBuzz(testCase.input);
    }

    /**
     * Compare results using simple equality.
     */
    compareResults(actual: any, expected: any): boolean {
        return actual === expected;
    }
}

// Create and run the driver
const driver = new BaseDriver(new FizzBuzzDriver());
driver.run();
