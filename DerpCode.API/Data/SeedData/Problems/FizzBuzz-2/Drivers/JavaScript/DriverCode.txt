import { BaseDriver, IProblemDriver } from './base-driver.js';
import { fizzBuzz } from './solution.js';

/**
 * Problem-specific driver for FizzBuzz problem.
 */
class FizzBuzzDriver extends IProblemDriver {
    /**
     * Parse input as array of integers and expected output as array of strings.
     */
    parseTestCases(input, expectedOutput) {
        const testCases = [];
        
        for (let i = 0; i < input.length; i++) {
            testCases.push({
                input: input[i],
                expectedOutput: expectedOutput[i]
            });
        }
        
        return testCases;
    }

    /**
     * Execute the fizzBuzz function with the test case input.
     */
    executeTestCase(testCase) {
        return fizzBuzz(testCase.input);
    }

    /**
     * Compare results using string equality.
     */
    compareResults(actual, expected) {
        return actual === expected;
    }

    /**
     * Format error message for failed tests.
     */
    formatErrorMessage(actual, expected) {
        return `Expected '${expected}' but got '${actual}'`;
    }
}

// Create and run the driver
const driver = new BaseDriver(new FizzBuzzDriver());
driver.run();
